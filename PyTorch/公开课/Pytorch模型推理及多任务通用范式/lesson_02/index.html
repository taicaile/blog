    import torch
import torch.onnx
from torchvision import models

DEVICE = torch.device("cpu")

dummy_input = torch.randn(1, 3, 224, 224).type(torch.float32).to(DEVICE)
model = models.resnet.resnet18(pretrained=False, progress=True)
model.load_state_dict(torch.load("weights/resnet18-5c106cde.pth"))
model.to(DEVICE)
model.eval()
torch.onnx.export(model, dummy_input, "resnet18.onnx")
