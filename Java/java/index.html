    /* 变量定义 */
// 定义变量，变量需要初始化之后才能使用
int a = 10;
// 输出变量
System.out.println(a);

/* 数据类型 */
// 数值默认是整数类型
long l = 10000L;
System.out.println(l);

// 浮点默认是 double 类型
float f = 13.14F;
System.out.println(f);

// 自动类型转换，数据范围从小到大
float d = 10;
System.out.println(d);

// 强制类型转换
int k = (int) 88.88;
System.out.println(k);
// 88

/* 算数运算符 */
int a1 = 10;
int b1 = 20;
System.out.println(a1 + b1);
System.out.println(a1 - b1);
System.out.println(a1 * b1);
// 整数相除只能得到整数
System.out.println(a1 / b1);
System.out.println((float) a1 / b1);
System.out.println(a1 % b1);

/* 字符的 + 操作 */
int i = 10;
char c = 'A'; // A 对应的数值是 65
// 表达式提升，表达结果自动提升表达式中最高的数据类型
System.out.println(i + c);

// 字符串加法，为字符串拼接
System.out.println("it" + "666");
System.out.println("it" + 666);
System.out.println(666 + "it");
System.out.println("it" + 666 + 888);
System.out.println(1 + 99 + "it"); // 注意这里输出为 100it

/* 类型转换 */
short s = 10;
s += 20;
// s = s + 20  // 注意这里会报错
s = (short)(s + 20);
// 20 默认是 int 类型，
System.out.println("s:" + s);
